#include <iostream>
using namespace std;
#define n 1000000
long long phi[1000001];
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0); cout.tie(0);
	for(int i=1;i<=n;i++)
	phi[i]=i;
	phi[1]=1;
	for(int i=2;i<=n;i++)
	{
		if(phi[i]==i){
			phi[i]=i-1;
		for(int j=2*i;j<=n;j+=i)
		{
			phi[j]=(phi[j]/i)*(i-1);
		}
		}
	}
	int t;
	cin>>t;
	while(t--)
	{
		int x;
		cin>>x;
		cout<<phi[x]<<endl;
	}
	return 0;
}
